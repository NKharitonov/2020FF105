#include <iostream>
using namespace std;

int NOD(int a, int b) {
    if (a < b) {
        swap(a, b);
    }
    while (a % b != 0) {
        a = a % b;
        swap(a, b);
    }
    return b;
}

 int NOK(int a, int b) {
    return (a * b) / NOD(a, b);
}

    cout << "НОК: "<<NOK(A, B) << endl;
}
